class Backpack.Component extends Backbone.View

  tagName: 'div'

  options:
    name:    ''
    content: ''
    parent:  'body'

  initialize: ->
    @

  render: =>
    @parent.append @el
    @

  before: (content) =>
    @setContent content
    @$el.before content
    @
  
  after: (content) =>
    @setContent content
    @$el.after @content
    @

  show: =>
    @delegateEvents @events
    @$el.removeClass 'hide'
    @

  hide: =>
    @$el.addClass 'hide'
    @undelegateEvents()
    @

  close: =>
    @hide()
    @$el.remove()
    @

  setContent: (content) =>
    if content.el?
      @content = content.render().el
    else
      @content = content
    @
  
  setParent: (parent) =>
    @parent = $(parent)
    @

  addClass: (klass) =>
    @$el.addClass klass
    @
